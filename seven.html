#include #include int myprintf(char*fmt,...) { char*p; va_list ap; int iVal; double dVal; va_start(ap,fmt); for(p=fmt;*p;p++) { if(*p!='%') { putchar(*p); continue; } switch(*++p) { case 'd': iVal=va_arg(ap,int); printf("%d",iVal); break; case 'f': dVal=va_arg(ap,double); printf("%1.2f",dVal); break; case's': for (sval=va_arg(ap,char*);*sval;sval++) putchar(*p); break; } } return 0; } void main() { myprintf("%s Integers: %d and fractions: %f and %k","check",1,1.91,1.56); }
